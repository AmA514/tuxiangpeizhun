6. 代码
	img1=imread('A.jpg');
	img2=imread('B.jpg');
	figure;
	subplot(1,2,1),imshow(img1);
	subplot(1,2,2),imshow(img2);
	img3=cp2tform(movingPoints1,fixedPoints1,'affine');
	img4=imtransform(img1,img3);
	figure
	subplot(1,2,1),imshow(img4);
	subplot(1,2,2),imshow(img2);
